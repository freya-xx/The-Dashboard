<script>
// 🔴 FIREBASE CONFIG
const firebaseConfig = { apiKey: "YOUR_KEY", authDomain: "YOUR_ID.firebaseapp.com", databaseURL: "https://YOUR_ID.firebaseio.com", projectId: "YOUR_ID", storageBucket: "YOUR_ID.appspot.com", messagingSenderId: "123", appId: "1:123" };

const STORAGE_KEY = 'medlife_hq_v3';
const defaultState = {
  tasks: [], expenses: [], journals: [], wellbeingLog: [], income: 5000, 
  financeTab: 'current', healthTab: 'checkin' 
};

let S = { ...defaultState };
let db; let useCloud = false;

if (firebaseConfig.apiKey !== "YOUR_KEY") {
  firebase.initializeApp(firebaseConfig); db = firebase.database(); useCloud = true;
  document.getElementById('sync-status').textContent = '🟢 Cloud Synced';
}

const $ = id => document.getElementById(id);
const uid = () => Date.now();
function toast(m) { const t=$('toast'); t.textContent=m; t.classList.add('show'); setTimeout(()=>t.classList.remove('show'),2000); }

// SAVE & LOAD
function save() {
  localStorage.setItem(STORAGE_KEY, JSON.stringify(S));
  if (useCloud) db.ref('data').set(S);
}

function init() {
  const local = localStorage.getItem(STORAGE_KEY);
  if (local) S = { ...defaultState, ...JSON.parse(local) };
  if (useCloud) {
    db.ref('data').on('value', snap => {
      if (snap.exists()) { S = snap.val(); renderAll(); }
    });
  }
  renderAll();
}

function renderAll() {
  renderHome(); renderFinance(); renderHealth();
}

// NAVIGATION
function goTo(p, el) {
  document.querySelectorAll('.page').forEach(x => x.classList.remove('active'));
  document.querySelectorAll('.nav-item').forEach(x => x.classList.remove('active'));
  $('page-'+p).classList.add('active'); if(el) el.classList.add('active');
}

function switchTab(page, tab, el) {
  const container = el.parentElement;
  container.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
  el.classList.add('active');

  if (page === 'finance') {
    S.financeTab = tab;
    $('finance-current-view').style.display = tab === 'current' ? 'block' : 'none';
    $('finance-history-view').style.display = tab === 'history' ? 'block' : 'none';
    renderFinance();
  } else if (page === 'health') {
    S.healthTab = tab;
    $('health-checkin-view').style.display = tab === 'checkin' ? 'block' : 'none';
    $('health-journal-view').style.display = tab === 'journal' ? 'block' : 'none';
    renderHealth();
  }
}

// HOME
function renderHome() {
  $('home-date').textContent = new Date().toLocaleDateString();
  const spent = S.expenses.length;
  $('home-stats').innerHTML = `<div class="stat-card"><div class="stat-val">${spent}</div><div class="stat-lbl">Expenses Logged</div></div>`;
}

// FINANCE
function addExpense() {
  const n = $('exp-name').value; const a = parseFloat($('exp-amt').value);
  if(!n || !a) return toast("Enter details");
  S.expenses.unshift({ id: uid(), name: n, amount: a, cat: $('exp-cat').value, date: new Date().toISOString() });
  $('exp-name').value = ''; $('exp-amt').value = '';
  save(); renderFinance(); toast("Saved");
}

function deleteExpense(id) {
  S.expenses = S.expenses.filter(e => e.id !== id);
  save(); renderFinance();
}

function renderFinance() {
  const now = new Date();
  const curM = now.getMonth(); const curY = now.getFullYear();
  
  // Grouping for History
  const monthlyData = {};
  S.expenses.forEach(e => {
    const d = new Date(e.date);
    const key = d.toLocaleString('default', { month: 'long', year: 'numeric' });
    if (!monthlyData[key]) monthlyData[key] = { list: [], total: 0 };
    monthlyData[key].list.push(e);
    monthlyData[key].total += e.amount;
  });

  // Render Current
  const currentExpenses = S.expenses.filter(e => {
    const d = new Date(e.date); return d.getMonth() === curM && d.getFullYear() === curY;
  });
  const spent = currentExpenses.reduce((a,b)=>a+b.amount,0);
  
  $('fin-stats').innerHTML = `
    <div class="stat-card"><div class="stat-val">₱${S.income}</div><div class="stat-lbl">Budget</div></div>
    <div class="stat-card"><div class="stat-val" style="color:var(--accent3)">₱${spent}</div><div class="stat-lbl">Spent</div></div>
    <div class="stat-card"><div class="stat-val" style="color:var(--accent)">₱${S.income-spent}</div><div class="stat-lbl">Left</div></div>
  `;

  $('expense-list').innerHTML = currentExpenses.map(e => `
    <div class="expense-row">
      <div class="expense-name">${e.name}<div style="font-size:10px; color:var(--hint)">${e.cat}</div></div>
      <div style="display:flex; align-items:center; gap:10px">
        <div style="font-weight:700">₱${e.amount}</div>
        <button class="del-btn" onclick="deleteExpense(${e.id})">×</button>
      </div>
    </div>
  `).join('');

  // Render History
  $('history-list').innerHTML = Object.keys(monthlyData).map(month => `
    <div class="history-month-header">${month} <span style="float:right">Total: ₱${monthlyData[month].total}</span></div>
    ${monthlyData[month].list.map(e => `<div class="expense-row" style="font-size:12px">${e.name} <span>₱${e.amount}</span></div>`).join('')}
  `).join('');
}

// WELLBEING
function updWB(s, id) { $(id).textContent = s.value + '/10'; }
function saveWB() {
  const log = { date: new Date().toLocaleDateString(), e: $('e-slider').value, s: $('s-slider').value, sl: $('sl-slider').value };
  const idx = S.wellbeingLog.findIndex(l => l.date === log.date);
  if(idx > -1) S.wellbeingLog[idx] = log; else S.wellbeingLog.push(log);
  save();
}

function saveJournal() {
  const txt = $('journal-text').value;
  if(!txt) return;
  S.journals.unshift({ id: uid(), text: txt, date: new Date().toLocaleString() });
  $('journal-text').value = '';
  save(); renderHealth(); toast("Journaled!");
}

function renderHealth() {
  $('full-journal-list').innerHTML = S.journals.map(j => `
    <div style="padding:15px; border-bottom:1px solid var(--border)">
      <div style="font-size:11px; color:var(--accent2); font-weight:700; margin-bottom:5px">${j.date}</div>
      <div style="font-size:13.5px; line-height:1.6">${j.text}</div>
    </div>
  `).join('') || '<div class="empty">No entries yet.</div>';
}

function toggleSidebar() { $('sidebar').classList.toggle('open'); }
function closeSidebar() { $('sidebar').classList.remove('open'); }

document.addEventListener('DOMContentLoaded', init);
</script>
