let activePageId = 'home';
function goTo(p, el) {

  document.querySelectorAll('.page').forEach(x => x.classList.remove('active'));

  $('page-'+p).classList.add('active');

  renderPage(p);
}

function switchSubTab(page, tab, el) {
  el.parentElement.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
  el.classList.add('active');
  if (page === 'finance') {
    $('finance-current-view').style.display = tab === 'current' ? 'block' : 'none';
    $('finance-history-view').style.display = tab === 'history' ? 'block' : 'none';
  } else if (page === 'health') {
    $('health-checkin-view').style.display = tab === 'checkin' ? 'block' : 'none';
    $('health-history-view').style.display = tab === 'history' ? 'block' : 'none';
  }
}

function renderPage(p) {
  if (p === 'home') renderHome();
  if (p === 'finance') renderFinance();
  if (p === 'health') renderHealth();
}

function renderHome() {
  $('home-date').textContent = new Date().toLocaleDateString('en-PH', { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' });
  $('home-stats').innerHTML = `<div class="stat-card"><div class="stat-val">${S.expenses.length}</div><div class="stat-lbl">Expenses Logged</div></div>`;
}

// Finance Logic
function addExpense() {
  const n = $('exp-name').value; const a = parseFloat($('exp-amt').value);
  if(!n || !a) return;
  S.expenses.unshift({ id: uid(), name: n, amount: a, cat: $('exp-cat').value, date: new Date().toISOString() });
  $('exp-name').value = ''; $('exp-amt').value = '';
  saveState(); renderFinance();
}

function deleteExpense(id) {
  S.expenses = S.expenses.filter(e => e.id !== id);
  saveState(); renderFinance();
}

function renderFinance() {
  const now = new Date();
  const monthlyData = {};
  S.expenses.forEach(e => {
    const d = new Date(e.date);
    const key = d.toLocaleString('default', { month: 'long', year: 'numeric' });
    if (!monthlyData[key]) monthlyData[key] = { list: [], total: 0 };
    monthlyData[key].list.push(e);
    monthlyData[key].total += e.amount;
  });

  const currentExpenses = S.expenses.filter(e => {
    const d = new Date(e.date); return d.getMonth() === now.getMonth() && d.getFullYear() === now.getFullYear();
  });
  const spent = currentExpenses.reduce((a,b)=>a+b.amount,0);

  $('fin-stats').innerHTML = `
    <div class="stat-card"><div class="stat-val">₱${S.income}</div><div class="stat-lbl">Allowance</div></div>
    <div class="stat-card"><div class="stat-val" style="color:var(--accent3)">₱${spent}</div><div class="stat-lbl">Spent</div></div>
    <div class="stat-card"><div class="stat-val" style="color:var(--accent)">₱${S.income-spent}</div><div class="stat-lbl">Left</div></div>
  `;

  $('expense-list').innerHTML = currentExpenses.map(e => `
    <div class="expense-row"><span>${e.name}</span><strong>₱${e.amount}</strong><button class="del-btn" onclick="deleteExpense(${e.id})">×</button></div>
  `).join('') || '<div class="empty">No expenses this month</div>';

  $('history-list').innerHTML = Object.keys(monthlyData).map(month => `
    <div class="history-month-header">${month} <span style="float:right">Total: ₱${monthlyData[month].total}</span></div>
    ${monthlyData[month].list.map(e => `<div class="expense-row">${e.name} <span>₱${e.amount}</span></div>`).join('')}
  `).join('') || '<div class="empty">No history found</div>';
}

// Health Logic
function updWB(s, id) { $(id).textContent = s.value + '/10'; }
function saveWB() {

  if(idx > -1) S.wellbeingLog[idx] = log; else S.wellbeingLog.push(log);
  saveState();
}

function saveJournal() {

  if(!txt) return;
  S.journals.unshift({ id: uid(), text: txt, date: new Date().toLocaleString('en-PH') });
  $('journal-text').value = '';

}

    </div>

}
