function goTo(id, el) {
  document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
  document.querySelectorAll('.nav-item').forEach(n => n.classList.remove('active'));
  $('page-' + id).classList.add('active');
  if (el) el.classList.add('active');
  closeSidebar();
  renderPage(id);
}

function renderPage(id) {
  const renders = {
    home: renderHome,
    rotations: renderRotations,
    study: renderStudy,
    exams: renderExams,
    schedule: renderSchedule,
    tasks: () => renderTasks(taskFilter),
    habits: renderHabits,
    health: renderHealth,
    finance: renderFinance,
  };
  if (renders[id]) renders[id]();
}

function toggleSidebar() {
  $('sidebar').classList.toggle('open');
  $('sidebar-backdrop').classList.toggle('open');
}
function closeSidebar() {
  $('sidebar').classList.remove('open');
  $('sidebar-backdrop').classList.remove('open');
}

function renderHome() {
  $('greeting-time').textContent = greeting();
  $('home-quote').textContent = QUOTES[today.getDay() % QUOTES.length];
  $('home-date-str').textContent = today.toLocaleDateString('en-PH', { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' });

  const doneTasks = S.tasks.filter(t => t.done).length;
  const doneHabits = S.habits.filter(h => h.log && h.log[6]).length;
  const totalStudyH = S.studyLog.reduce((a, s) => a + (parseFloat(s.hours) || 0), 0);
  const clerkDate = new Date('2025-07-15');
  const daysLeft = Math.max(0, Math.ceil((clerkDate - today) / (1000 * 60 * 60 * 24)));

  $('home-stats').innerHTML = [
    { val: `${doneTasks}/${S.tasks.length}`, lbl: 'Tasks done', color: 'var(--accent)' },
    { val: `${doneHabits}/${S.habits.length}`, lbl: 'Habits tracked', color: 'var(--accent4)' },
    { val: daysLeft + 'd', lbl: 'Days to clerkship', color: 'var(--accent2)' },
    { val: totalStudyH.toFixed(1) + 'h', lbl: 'Study hrs logged', color: 'var(--accent3)' },
  ].map(s => `<div class="stat-card"><div class="stat-val" style="color:${s.color}">${s.val}</div><div class="stat-lbl">${s.lbl}</div></div>`).join('');

  const pending = S.tasks.filter(t => !t.done).slice(0, 5);
  $('home-tasks').innerHTML = pending.length
    ? pending.map(t => `<div class="task-item"><div class="task-text">${t.text}</div><span class="badge badge-${CAT_COLORS[t.cat]||'blue'}">${t.cat}</span></div>`).join('')
    : '<div class="empty"><div class="empty-icon">✓</div>All caught up!</div>';

  const events = S.events.slice().sort((a, b) => a.start.localeCompare(b.start)).slice(0, 5);
  $('home-events').innerHTML = events.length
    ? events.map(e => `<div style="display:flex;align-items:center;gap:8px;padding:6px 0;border-bottom:1px solid var(--border)"><span style="font-size:11.5px;color:var(--muted);font-family:var(--syne);font-weight:600;min-width:48px">${e.start}</span><span style="font-size:13.5px">${e.name}</span></div>`).join('')
    : '<div class="empty"><div class="empty-icon">📅</div>No events yet</div>';

  const goal = S.studyWeekGoal || 20;
  const pct = Math.min(100, Math.round((totalStudyH / goal) * 100));
  $('home-study-goal-text').innerHTML = `<span>Study progress</span><span style="font-weight:600">${totalStudyH.toFixed(1)}h / ${goal}h goal</span>`;
  $('home-study-bar').style.width = pct + '%';
}

function addRotation() {
  const name = $('r-name').value.trim();
  if (!name) { toast('Enter a rotation name'); return; }
  S.rotations.push({ id: uid(), name, start: $('r-start').value, end: $('r-end').value, type: $('r-type').value });
  $('r-name').value = '';
  saveState(); closeModal('rot-modal'); renderRotations(); toast('Rotation added!');
}
function deleteRotation(id) {
  S.rotations = S.rotations.filter(r => r.id !== id);
  saveState(); renderRotations();
}
function addProc() {
  const name = $('p-name').value.trim();
  if (!name) { toast('Enter a procedure name'); return; }
  S.procedures.push({ id: uid(), name, level: $('p-level').value, notes: $('p-notes').value, date: today.toLocaleDateString('en-PH') });
  $('p-name').value = ''; $('p-notes').value = '';
  saveState(); closeModal('proc-modal'); renderRotations(); toast('Procedure logged!');
}
function deleteProc(id) {
  S.procedures = S.procedures.filter(p => p.id !== id);
  saveState(); renderRotations();
}

function renderRotations() {

  let weeks = '—', pct = 0;
  if (cur && cur.start && cur.end) {
    const s = new Date(cur.start), e = new Date(cur.end);
    const totalW = Math.ceil((e - s) / (7 * 86400000));
    const doneW = Math.max(0, Math.ceil((today - s) / (7 * 86400000)));
    weeks = `${Math.min(doneW, totalW)} of ${totalW}`;
    pct = Math.min(100, Math.round((doneW / totalW) * 100));
  }
